Data Contract

推荐 JSON 数据格式

下面是建议的中间文件格式，方便后续脚本、LLM API 和前端页面连接。

paper_dict.json

{
  "title": "Paper Title",
  "authors": ["Author A", "Author B"],
  "abstract": "...",
  "sections": [
    {
      "id": "sec_1",
      "title": "Introduction",
      "content": "...",
      "figures": [],
      "equations": []
    }
  ],
  "references": []
}

outline.json

{
  "talk_title": "Video Title",
  "duration": 180,
  "chapters": [
    {
      "id": "ch_1",
      "title": "Motivation",
      "goal": "...",
      "source_sections": ["sec_1"]
    }
  ]
}

storyboard.json

{
  "slides": [
    {
      "slide_id": "slide_001",
      "title": "Motivation",
      "visual_type": "bullet",
      "latex": "...",
      "narration": "...",
      "cursor": [
        {
          "time": 1.2,
          "x": 0.42,
          "y": 0.31,
          "action": "point"
        }
      ],
      "subtitle": [
        {
          "start": 0,
          "end": 3.5,
          "text": "..."
        }
      ]
    }
  ]
}
